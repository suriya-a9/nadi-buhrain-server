db.createView(
  "request_with_names",
  "reques",
  [
    {
      $lookup: {
        from: "useraccounts",
        localField: "senderId",
        foreignField: "_id",
        as: "sender"
      }
    },
    { $unwind: { path: "$sender", preserveNullAndEmptyArrays: true } },
    {
      $lookup: {
        from: "useraccounts",
        localField: "receiverId",
        foreignField: "_id",
        as: "receiver"
      }
    },
    { $unwind: { path: "$receiver", preserveNullAndEmptyArrays: true } },
    {
      $project: {
        request: 1,
        points: 1,
        status: 1,
        reason: 1,
        createdAt: 1,
        updatedAt: 1,
        senderName: "$sender.basicInfo.fullName",
        receiverName: "$receiver.basicInfo.fullName"
      }
    }
  ]
)
